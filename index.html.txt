// Main Function for Handling Date Input
function getAstroDetails(dateOfBirth) {
    const birthDate = new Date(dateOfBirth);
    const day = birthDate.getDate();
    const month = birthDate.getMonth() + 1; // Months are 0-indexed
    const year = birthDate.getFullYear();

    // Zodiac Sign Calculation
    const zodiacSign = getZodiacSign(day, month);

    // Deterministic Lucky Number (based on date sum)
    const luckyNumber = calculateLuckyNumber(day, month, year);

    // Element based on Zodiac Sign (consistently mapped)
    const element = getZodiacElement(zodiacSign);

    return {
        zodiacSign,
        luckyNumber,
        element,
    };
}

// Helper Functions
function getZodiacSign(day, month) {
    const zodiacSigns = [
        { sign: "Capricorn", start: [12, 22], end: [1, 19] },
        { sign: "Aquarius", start: [1, 20], end: [2, 18] },
        { sign: "Pisces", start: [2, 19], end: [3, 20] },
        { sign: "Aries", start: [3, 21], end: [4, 19] },
        { sign: "Taurus", start: [4, 20], end: [5, 20] },
        { sign: "Gemini", start: [5, 21], end: [6, 20] },
        { sign: "Cancer", start: [6, 21], end: [7, 22] },
        { sign: "Leo", start: [7, 23], end: [8, 22] },
        { sign: "Virgo", start: [8, 23], end: [9, 22] },
        { sign: "Libra", start: [9, 23], end: [10, 22] },
        { sign: "Scorpio", start: [10, 23], end: [11, 21] },
        { sign: "Sagittarius", start: [11, 22], end: [12, 21] },
    ];

    return zodiacSigns.find(z =>
        (month === z.start[0] && day >= z.start[1]) ||
        (month === z.end[0] && day <= z.end[1])
    ).sign;
}

function calculateLuckyNumber(day, month, year) {
    // Consistent lucky number derived from sum of digits of DOB
    const sum = `${day}${month}${year}`
        .split('')
        .map(Number)
        .reduce((acc, num) => acc + num, 0);

    return sum % 9 || 9; // Return 1-9, fallback to 9
}

function getZodiacElement(zodiacSign) {
    const elementMap = {
        "Aries": "Fire",
        "Leo": "Fire",
        "Sagittarius": "Fire",
        "Taurus": "Earth",
        "Virgo": "Earth",
        "Capricorn": "Earth",
        "Gemini": "Air",
        "Libra": "Air",
        "Aquarius": "Air",
        "Cancer": "Water",
        "Scorpio": "Water",
        "Pisces": "Water",
    };

    return elementMap[zodiacSign];
}

// Event Listener to Display Results
document.getElementById("submit").addEventListener("click", () => {
    const dob = document.getElementById("dob").value;
    const details = getAstroDetails(dob);

    document.getElementById("result").innerHTML = `
        <p>Zodiac Sign: ${details.zodiacSign}</p>
